def self.graph(graph_name, opt={}, &block)
  parser = if opt[:ruby_version] == :ruby18
    Ruby18Parser.new
  else
    Ruby19Parser.new
  end

  GraphBuilder.new(graph_name, parser).
    define(&block).
    find_nodes.
    find_relations.
    finalize.
    render
end

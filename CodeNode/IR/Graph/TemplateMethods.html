def each_class(&block)

    node.class?
  end.sort.each &block
end

def each_containment(&block)
  @nodes.values.sort.each do |node|
    if node.parent
      block.call node.parent, node
    end
  end
end

def each_inheritance(&block)
  @nodes.values.sort.each do |node|
    if node.super_class_node
      block.call node, node.super_class_node
    end
  end
end
